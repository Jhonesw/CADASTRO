    function configurarLimiteIdade() {
        const hoje = new Date();
        const dataLimite = new Date(hoje.getFullYear() - 10, hoje.getMonth(), hoje.getDate());
        const minDate = dataLimite.toISOString().split("T")[0];
        document.getElementById('nascimento').setAttribute('min', minDate);
    }

    async function salvar() {
        const foto = document.getElementById('foto').files[0];
        const bairroSel = document.getElementById('bairro').value;
        const msgBairro = mensagensBairro[bairroSel] || "Cadastro realizado com sucesso.";

    async function alterar() {
        const id = parseInt(document.getElementById('edit-id').value);
        const fotoInput = document.getElementById('foto').files[0];
        
        const registro = {
            id: id,
            crianca: document.getElementById('crianca').value,

            rg: document.getElementById('rg').value,

    function excluir() {
        if(confirm("Tem certeza que deseja excluir este cadastro?")) {
            const id = parseInt(document.getElementById('edit-id').value);
            const tx = db.transaction("pessoas", "readwrite");
            tx.objectStore("pessoas").delete(id);
            tx.oncomplete = () => {
                alert("Cadastro removido!");
                showTab('lista');
            };
        }
    }

    const toBase64 = f => new Promise(r => {
        const rd = new FileReader(); rd.onload = () => r(rd.result); rd.readAsDataURL(f);
    });

    function renderLista() {
        const store = db.transaction("pessoas").objectStore("pessoas");
        store.getAll().onsuccess = (e) => {
            todosCadastros = e.target.result;
            displayCadastros(todosCadastros);
        };
    }

    function displayCadastros(itens) {
        const cont = document.getElementById('lista-db');
        cont.innerHTML = "";
        itens.forEach(d => {
            const div = document.createElement('div');
            div.className = 'lista-item';
            div.onclick = () => abrir(d);
            div.innerHTML = `<img src="${d.foto || 'https://via.placeholder.com/50'}">
                <div><b>${d.crianca}</b><br><small>${d.bairro || 'Sem Bairro'} | RG: ${d.rg}</small></div>`;
            cont.appendChild(div);
        });
    }

    function filtrarLista() {
        const termo = document.getElementById('inputBusca').value.toLowerCase();
        const filtrados = todosCadastros.filter(p => 
            p.crianca.toLowerCase().includes(termo) || 
            p.rg.includes(termo) || 
            p.bairro.toLowerCase().includes(termo)
        );
        displayCadastros(filtrados);
    }

    function exportarBackup() {
        db.transaction("pessoas").objectStore("pessoas").getAll().onsuccess = (e) => {
            const blob = new Blob([JSON.stringify(e.target.result)], {type: "application/json"});
            const a = document.createElement('a');
            a.href = URL.createObjectURL(blob);
            a.download = `backup_comunidade_${new Date().getTime()}.json`;
            a.click();
        };
    }

    function importarBackup(input) {
        const reader = new FileReader();
        reader.onload = () => {
            const dados = JSON.parse(reader.result);
            const tx = db.transaction("pessoas", "readwrite");
            const store = tx.objectStore("pessoas");
            dados.forEach(i => { delete i.id; store.add(i); });
            tx.oncomplete = () => { alert("Importado!"); renderLista(); showTab('lista'); };
        };
        reader.readAsText(input.files[0]);
    }
